    <script>
        // el body es el cuerpo del html donde se define todo
        // script es el codigo de js que sirve para html, super util para conseguir datos del back
        
        // funcion para conseguir data
        function fetchData() {
            fetch('/data') // fetch es una funcion de js que consigue datos, en este caso va a la ruta /data que se define en flask para conseguir los datos del csv
                .then(response => response.text()) // aqui pasa los datos a texto
                .then(data => {
                    document.getElementById('csv-table').innerHTML = data; //aqui inserta los datos en el contendor div, con el id csv-table
                });
        }

        // Fetch data every 5 seconds
        setInterval(fetchData, 5000); // ejecuta la funcion cada 5 segundos
        // Initial fetch
        fetchData();
    </script>
